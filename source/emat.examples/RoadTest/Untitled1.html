scope_file = emat.package_file("model", "tests", "road_test_nonuniform.yaml")
scp = Scope(scope_file)
exp_def = scp.design_experiments(
    n_samples_per_factor=1000,
    random_seed=1234,
    sampler='lhs',
)

assert len(exp_def) == scp.n_sample_factors() * 1000
